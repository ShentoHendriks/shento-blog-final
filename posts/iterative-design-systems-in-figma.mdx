// Example usage with CSS variables
<InteractivePlayground
  elementName="card"
  cssVariableScope=":root"
  options={[
    {
      name: "theme",
      label: "Theme",
      type: "select",
      property: "--primary-color",
      options: ["#3b82f6", "#10b981", "#f59e0b", "#ef4444"],
      default: "#3b82f6"
    },
    {
      name: "spacing",
      label: "Spacing",
      type: "dropdown",
      property: "--spacing",
      options: ["8px", "16px", "24px", "32px"],
      default: "16px"
    },
    {
      name: "borderStyle",
      label: "Border Style",
      type: "select",
      property: "--border-style",
      options: ["none", "solid", "dashed", "dotted"],
      default: "solid"
    },
    {
      name: "radius",
      label: "Border Radius",
      type: "slider",
      property: "--border-radius",
      min: 0,
      max: 50,
      default: 8,
      unit: "px"
    }
  ]}
  customCSS={`
    .card {
      background: var(--primary-color);
      padding: var(--spacing);
      border: 2px var(--border-style) rgba(0,0,0,0.2);
      border-radius: var(--border-radius);
      color: white;
    }
  `}
>
  <div className="card">
    <h3>Dynamic Card</h3>
    <p>This card uses CSS variables that you can control!</p>
  </div>
</InteractivePlayground>
